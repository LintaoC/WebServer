--- 1. Simple GET Request
echo -ne "GET / HTTP/1.1\r\nHost: localhost\r\nUser-Agent: curl/7.58.0\r\nAccept: */*\r\n\r\n" | nc localhost 8080

GET / HTTP/1.1
Host: localhost
User-Agent: curl/7.58.0
Accept: */*

--- 2. POST Request with URL-Encoded Form Data
echo -ne "POST /submit-form HTTP/1.1\r\nHost: localhost\r\nContent-Type: application/x-www-form-urlencoded\r\nContent-Length: 27\r\n\r\nusername=johndoe&password=1234" | nc localhost 8080

POST /submit-form HTTP/1.1
Host: localhost
Content-Type: application/x-www-form-urlencoded
Content-Length: 27

username=johndoe&password=1234

--- 3. HEAD Request (This is similar to GET but the server should only send back headers.)
echo -ne "HEAD / HTTP/1.1\r\nHost: localhost\r\nAccept: */*\r\n\r\n" | nc localhost 8080

Copy code
HEAD / HTTP/1.1
Host: localhost
Accept: */*

--- 4. POST Request with JSON Body

echo -ne "POST /api/data HTTP/1.1\r\nHost: localhost\r\nContent-Type: application/json\r\nContent-Length: 49\r\n\r\n{\"name\": \"John Doe\", \"age\": 30, \"city\": \"New York\"}" | nc localhost 8080

POST /api/data HTTP/1.1
Host: localhost
Content-Type: application/json
Content-Length: 49

{"name": "John Doe", "age": 30, "city": "New York"}

--- 5. GET Request with Query Parameters
echo -ne "GET /search?query=netcat&results=10 HTTP/1.1\r\nHost: localhost\r\nUser-Agent: curl/7.58.0\r\nAccept: */*\r\n\r\n" | nc localhost 8080

GET /search?query=netcat&results=10 HTTP/1.1
Host: localhost
User-Agent: curl/7.58.0
Accept: */*
